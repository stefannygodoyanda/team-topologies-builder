// ─── Estado Central ───────────────────────────────────────────────────
const TYPES = {

  undef:   { label:'Undefined',           fill:'var(--tt-undef)',    stroke:'var(--tt-undef-b)',    shape:'h', dashed:true },
};
const INTER = {
  collab:{ label:'Collaboration', stroke:'#967EE2', fill:'rgba(198,190,223,0.45)', dash:'8 4', marker:'arr-collab' },
  facil: { label:'Facilitation',  stroke:'#78996B', fill:'rgba(201,223,190,0.45)', dash:'8 4', marker:'arr-facil'  },
  xaas:  { label:'X-as-a-Service',stroke:'#999696', fill:'rgba(180,180,180,0.5)',  dash:'0',   marker:'arr-xaas'   },
};

let teams=[], conns=[], nid=1;
let selId=null, selConnId=null, dragType=null;
let connMode=null, connFrom=null;
let ctxId=null;
let wStep=1, wData={}, editMode=false, editTarget=null;

// Variables de vista (Mural-like zoom/pan)
let scale = 1, panX = 0, panY = 0;
let isPanning = false, startPanX, startPanY;

const svg   = document.getElementById('canvas');
const workspace = document.getElementById('workspace');
const tl    = document.getElementById('team-layer');
const cl    = document.getElementById('conn-layer');
const wrap  = document.getElementById('canvasWrap');

// ─── Zoom & Pan (Estilo Mural) ────────────────────────────────────────
wrap.addEventListener('mousedown', e => {
  if (e.target === svg || e.target === wrap) {
    isPanning = true;
    startPanX = e.clientX - panX;
    startPanY = e.clientY - panY;
  }
});
window.addEventListener('mousemove', e => {
  if (isPanning) {
    panX = e.clientX - startPanX;
    panY = e.clientY - startPanY;
    updateTransform();
  }
});
window.addEventListener('mouseup', () => { isPanning = false; });
wrap.addEventListener('wheel', e => {
  e.preventDefault();
  const zoomIntensity = 0.002;
  const wheel = e.deltaY < 0 ? 1 : -1;
  const zoom = Math.exp(wheel * zoomIntensity * 50);
  const newScale = Math.min(Math.max(0.15, scale * zoom), 4);
  
  const rect = wrap.getBoundingClientRect();
  const mouseX = e.clientX - rect.left;
  const mouseY = e.clientY - rect.top;
  
  panX = mouseX - (mouseX - panX) * (newScale / scale);
  panY = mouseY - (mouseY - panY) * (newScale / scale);
  scale = newScale;
  updateTransform();
}, {passive: false});

function updateTransform() {
  workspace.setAttribute('transform', `translate(${panX},${panY}) scale(${scale})`);
}
function resetView() {
  scale = 1; panX = 0; panY = 0; updateTransform();
}

// Transformar coordenadas de pantalla a Canvas SVG
function getWorkspaceCoords(clientX, clientY) {
  const r = wrap.getBoundingClientRect();
  return {
    x: (clientX - r.left - panX) / scale,
    y: (clientY - r.top - panY) / scale
  };
}

// ─── Drag from sidebar ────────────────────────────────────────────────
function startDrag(e, type) { dragType = type; e.dataTransfer.effectAllowed = 'copy'; }
wrap.addEventListener('dragover', e => e.preventDefault());
wrap.addEventListener('drop', e => {
  e.preventDefault();
  if (!dragType) return;
  const coords = getWorkspaceCoords(e.clientX, e.clientY);
  openWizard(dragType, coords.x, coords.y);
  dragType = null;
});

function renderWiz() {

  document.getElementById('btn-back').style.display = wStep > 1 ? '' : 'none';

  if(wStep===4) nb.textContent='Crear equipo ✓'; else if(editMode&&wStep===3) nb.textContent='Guardar ✓'; else nb.textContent='Siguiente →';

function nav(dir) {
  if (dir === 1) {
    if (wStep === 3) {
      wData.name = document.getElementById('wiz-name').value.trim() || 'Equipo sin nombre';
      wData.resp = document.getElementById('wiz-resp').value.trim(); wData.size = document.getElementById('wiz-size').value.trim();

}
function applyEdit() { editTarget.name = wData.name; editTarget.resp = wData.resp; editTarget.size = wData.size; rerender(editTarget); }

// ─── Render Teams ─────────────────────────────────────────────────────
function dims(type) {
  if(type==='enable') return {w:70,h:110}; if(type==='complex') return {w:130,h:85}; return {w:165,h:62};
}
function createTeam() {
  const r = wrap.getBoundingClientRect();
  const px = wData.px ?? ((r.width/2 - panX)/scale);
  const py = wData.py ?? ((r.height/2 - panY)/scale);
  const {w,h} = dims(wData.type);
  const t = { id: nid++, type: wData.type, name: wData.name||'Equipo '+nid, resp: wData.resp, size: wData.size, comment: '', x: px - w/2, y: py - h/2 };
  teams.push(t); rerender(t); countUpdate(); toast('Equipo creado');
}

function rerender(t) {
  const old = document.getElementById('t-'+t.id); if (old) old.remove();
  const T = TYPES[t.type] || TYPES.undef; const { w, h } = dims(t.type);
  const g = mkSvg('g'); g.id = 't-' + t.id; g.setAttribute('transform', `translate(${t.x},${t.y})`); g.style.cursor = 'pointer';

  let shape;
  if (t.type === 'complex') {
    shape = mkSvg('polygon');
    shape.setAttribute('points', `${w*.28},0 ${w*.72},0 ${w},${h*.28} ${w},${h*.72} ${w*.72},${h} ${w*.28},${h} 0,${h*.72} 0,${h*.28}`);
  } else {
    shape = mkSvg('rect'); shape.setAttribute('width', w); shape.setAttribute('height', h); shape.setAttribute('rx', '9');
  }
  shape.setAttribute('fill', T.fill); shape.setAttribute('stroke', T.stroke); shape.setAttribute('stroke-width', '2');
  if (T.dashed) shape.setAttribute('stroke-dasharray', '7 3');
  g.appendChild(shape);

  const isV = t.type === 'enable';
  const lbl = mkSvg('text'); lbl.setAttribute('x', w/2); lbl.setAttribute('y', h/2 - (isV?5:6));
  lbl.setAttribute('text-anchor', 'middle'); lbl.setAttribute('dominant-baseline', 'central');
  lbl.setAttribute('font-family', 'Syne,sans-serif'); lbl.setAttribute('font-size', isV?'11':'12');
  lbl.setAttribute('font-weight', '600'); lbl.setAttribute('fill', 'rgba(0,0,0,0.82)'); lbl.setAttribute('pointer-events', 'none');
  if (isV) lbl.setAttribute('transform', `rotate(-90,${w/2},${h/2})`);
  lbl.textContent = trunc(t.name, isV ? 10 : 18); g.appendChild(lbl);

  // Comment UI (Post-it note visual)
  if (t.comment && t.comment.trim() !== '') {
    const comGrp = mkSvg('g');
    comGrp.setAttribute('transform', `translate(${w/2 - 50}, ${h + 10})`);
    const bg = mkSvg('rect'); bg.setAttribute('width', '100'); bg.setAttribute('height', '24'); bg.setAttribute('rx', '4'); bg.setAttribute('fill', '#f1c40f'); bg.style.boxShadow="var(--shadow)";
    const txt = mkSvg('text'); txt.setAttribute('x', '50'); txt.setAttribute('y', '16'); txt.setAttribute('text-anchor', 'middle'); txt.setAttribute('font-family', 'Syne,sans-serif'); txt.setAttribute('font-size', '9'); txt.setAttribute('font-weight','600'); txt.setAttribute('fill', '#000');
    txt.textContent = '💬 ' + trunc(t.comment, 14);
    comGrp.appendChild(bg); comGrp.appendChild(txt); g.appendChild(comGrp);
  }

  // Events
  let drag=false, moved=false, sTx=0, sTy=0, mOx=0, mOy=0;
  g.addEventListener('mousedown', e => {
    if (e.button !== 0) return;
    if (connMode && connFrom !== t.id) { addConn(connFrom, t.id, connMode); endConn(); return; }
    drag = true; moved = false; e.stopPropagation();
    mOx = e.clientX; mOy = e.clientY; sTx = t.x; sTy = t.y;
  });
  g.addEventListener('dblclick', () => openEdit(t));
  g.addEventListener('contextmenu', e => { e.preventDefault(); showCtx(e, t.id); });

  const mm = e => {
    if (!drag) return;
    const dx = (e.clientX - mOx) / scale; const dy = (e.clientY - mOy) / scale;
    if (Math.abs(dx)>1||Math.abs(dy)>1) moved = true;
    t.x = sTx + dx; t.y = sTy + dy;
    g.setAttribute('transform', `translate(${t.x},${t.y})`); drawConns();
  };
  const mu = e => {
    if (!drag) return; drag = false;
    if (!moved && e.button===0 && !connMode) { selConnId=null; selectTeam(t.id); }
  };
  document.addEventListener('mousemove', mm); document.addEventListener('mouseup', mu);
  tl.appendChild(g);
}

function center(t) { const {w,h} = dims(t.type); return {x: t.x+w/2, y: t.y+h/2}; }
function trunc(s,n) { return s.length>n ? s.slice(0,n)+'…' : s; }
function mkSvg(tag) { return document.createElementNS('http://www.w3.org/2000/svg', tag); }

function startConn(mode) { if(!selId){toast('Selecciona un equipo primero');return;} connMode=mode; connFrom=selId; document.getElementById('conn-info').classList.add('show'); document.body.style.cursor='crosshair'; }

function drawConns() {
  cl.innerHTML = '';
  conns.forEach(c => {
    const ft = teams.find(t=>t.id===c.from), tt = teams.find(t=>t.id===c.to); if(!ft||!tt) return;
    const fc = center(ft), tc = center(tt);
    const I = INTER[c.mode]; const mx=(fc.x+tc.x)/2, my=(fc.y+tc.y)/2, dx=tc.x-fc.x, dy=tc.y-fc.y;
    const len=Math.sqrt(dx*dx+dy*dy)||1, ux=dx/len, uy=dy/len, px=-uy, py=ux;
    
    const cg = mkSvg('g'); cg.style.cursor = 'pointer';
    const isSel = (c.id === selConnId);
    
    // Hitbox invisible gruesa para clics fáciles
    const hitbox = mkSvg('line'); hitbox.setAttribute('x1',fc.x); hitbox.setAttribute('y1',fc.y); hitbox.setAttribute('x2',tc.x); hitbox.setAttribute('y2',tc.y); hitbox.setAttribute('stroke','transparent'); hitbox.setAttribute('stroke-width','20'); cg.appendChild(hitbox);

    const activeStroke = isSel ? 'var(--sel-color)' : I.stroke;
    
    if (c.mode==='collab') {
      const sk=14; const poly=mkSvg('polygon'); poly.setAttribute('points',`${fc.x+px*sk},${fc.y+py*sk} ${tc.x+px*sk},${tc.y+py*sk} ${tc.x-px*sk},${tc.y-py*sk} ${fc.x-px*sk},${fc.y-py*sk}`);
      poly.setAttribute('fill',I.fill); poly.setAttribute('stroke',activeStroke); poly.setAttribute('stroke-width',isSel?'3':'1.5'); poly.setAttribute('stroke-dasharray',I.dash); poly.style.opacity='.75'; cg.appendChild(poly);
    } else if (c.mode==='facil') {
      const line=mkSvg('line'); line.setAttribute('x1',fc.x); line.setAttribute('y1',fc.y); line.setAttribute('x2',tc.x); line.setAttribute('y2',tc.y); line.setAttribute('stroke',activeStroke); line.setAttribute('stroke-width',isSel?'3':'1.5'); line.setAttribute('stroke-dasharray','7 4'); line.style.opacity='.55'; cg.appendChild(line);
      const circ=mkSvg('circle'); circ.setAttribute('cx',mx); circ.setAttribute('cy',my); circ.setAttribute('r','20'); circ.setAttribute('fill',I.fill); circ.setAttribute('stroke',activeStroke); circ.setAttribute('stroke-width',isSel?'3':'1.5'); circ.setAttribute('stroke-dasharray','7 4'); circ.style.opacity='.75'; cg.appendChild(circ);
    } else {
      const line=mkSvg('line'); line.setAttribute('x1',fc.x); line.setAttribute('y1',fc.y); line.setAttribute('x2',tc.x); line.setAttribute('y2',tc.y); line.setAttribute('stroke',activeStroke); line.setAttribute('stroke-width',isSel?'3':'1.5'); if(!isSel) line.setAttribute('marker-end',`url(#${I.marker})`); line.style.opacity='.65'; cg.appendChild(line);
      const tip={x:mx+ux*18,y:my+uy*18}, b1={x:mx-px*12-ux*10,y:my-py*12-uy*10}, b2={x:mx+px*12-ux*10,y:my+py*12-uy*10};
      const tri=mkSvg('polygon'); tri.setAttribute('points',`${tip.x},${tip.y} ${b1.x},${b1.y} ${b2.x},${b2.y}`); tri.setAttribute('fill',I.fill); tri.setAttribute('stroke',activeStroke); tri.setAttribute('stroke-width',isSel?'3':'1.5'); tri.style.opacity='.8'; cg.appendChild(tri);
    }
    const lbl=mkSvg('text'); lbl.setAttribute('x',mx); lbl.setAttribute('y',my-25); lbl.setAttribute('text-anchor','middle'); lbl.setAttribute('font-family','DM Mono,monospace'); lbl.setAttribute('font-size','9'); lbl.setAttribute('fill',activeStroke); lbl.style.opacity='.9'; lbl.textContent=I.label; cg.appendChild(lbl);
    
    cg.addEventListener('mousedown', e => { e.stopPropagation(); selectConn(c.id); });
    cl.appendChild(cg);
  });
}

function saveJSON() {
  const state = { teams, conns, scale, panX, panY };
  const blob = new Blob([JSON.stringify(state, null, 2)], { type: "application/json" });
  const a = document.createElement("a"); a.href = URL.createObjectURL(blob); a.download = "team-topologies-proyecto.json"; a.click();
  toast('Proyecto guardado (JSON)');
}

function loadJSON(e) {
  const file = e.target.files[0]; if (!file) return;
  const reader = new FileReader();
  reader.onload = ev => {
    try {
      const state = JSON.parse(ev.target.result);
      teams = state.teams || []; conns = state.conns || [];
      scale = state.scale || 1; panX = state.panX || 0; panY = state.panY || 0;
      nid = Math.max(0, ...teams.map(t=>t.id), ...conns.map(c=>c.id)) + 1;
      
      tl.innerHTML = ''; cl.innerHTML = ''; deselectTeams(); selConnId = null;
      teams.forEach(t => rerender(t)); drawConns(); countUpdate(); updateTransform();
      toast('Proyecto cargado correctamente');
    } catch(err) { toast('Error al leer el archivo JSON'); }
  };
  reader.readAsText(file);
  e.target.value = ''; // Reset input
}

function exportPNG() {
  // Reset zoom para la foto
  const oldScale = scale, oldPx = panX, oldPy = panY; resetView();
  deselectTeams(); selConnId = null; drawConns();
  
  setTimeout(() => {
    const s = document.getElementById('canvas');
    const xml = new XMLSerializer().serializeToString(s);
    const blob = new Blob([xml], {type: 'image/svg+xml;charset=utf-8'});
    const URLObject = window.URL || window.webkitURL || window;
    const blobURL = URLObject.createObjectURL(blob);
    
    const img = new Image();
    img.onload = () => {
      const canvas = document.createElement('canvas');
      const rect = wrap.getBoundingClientRect();
      canvas.width = rect.width; canvas.height = rect.height;
      const ctx = canvas.getContext('2d');
      // Fondo blanco forzado
      ctx.fillStyle = '#ffffff'; ctx.fillRect(0, 0, canvas.width, canvas.height);
      ctx.drawImage(img, 0, 0);
      
      // Añadir marca de agua
      ctx.font = '800 42px Syne, sans-serif'; ctx.fillStyle = 'rgba(0,0,0,0.05)';
      ctx.fillText('SGODOY', 40, canvas.height - 30);

      const a = document.createElement('a'); a.download = 'team-topologies.png'; a.href = canvas.toDataURL('image/png'); a.click();

      // Restaurar vista

    };

  }, 100);
}
